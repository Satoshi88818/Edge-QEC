class SumTree:
    """Sum-tree for efficient prioritized experience replay."""
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.tree = np.zeros(2 * capacity - 1)
        self.data = np.zeros(capacity, dtype=object)
        self.write_pointer = 0
        self.n_entries = 0

    def _propagate(self, idx: int, change: float) -> None:
        parent = (idx - 1) // 2
        self.tree[parent] += change
        if parent != 0:
            self._propagate(parent, change)

    def _retrieve(self, idx: int, s: float) -> int:
        left = 2 * idx + 1
        right = left + 1
        if left >= len(self.tree):
            return idx
        if s <= self.tree[left]:
            return self._retrieve(left, s)
        return self._retrieve(right, s - self.tree[left])

    def total(self) -> float:
        return self.tree[0]

    def append(self, priority: float, data: Tuple) -> None:
        idx = self.write_pointer + self.capacity - 1
        self.data[self.write_pointer] = data
        self.update(idx, priority)
        self.write_pointer = (self.write_pointer + 1) % self.capacity
        self.n_entries = min(self.n_entries + 1, self.capacity)

    def update(self, idx: int, priority: float) -> None:
        change = priority - self.tree[idx]
        self.tree[idx] = priority
        self._propagate(idx, change)

    def get(self, s: float) -> Tuple[int, float, Tuple]:
        idx = self._retrieve(0, s)
        data_idx = idx - self.capacity + 1
        return idx, self.tree[idx], self.data[data_idx]

    def __len__(self) -> int:
        return self.n_entries

class PriorityReplayBuffer:
    """Prioritized experience replay buffer using sum-tree."""
    def __init__(self, capacity: int):
        self.tree = SumTree(capacity)
        self.capacity = capacity
        self.max_priority = 1.0

    def append(self, priority: float, experience: Tuple) -> None:
        """Add experience with priority."""
        self.max_priority = max(self.max_priority, priority)
        self.tree.append(priority, experience)

    def sample(self, batch_size: int, alpha: float = CONFIG['per_alpha'], beta: float = CONFIG['per_beta']) -> Tuple[List, np.ndarray, List]:
        """Sample a batch with prioritized probabilities."""
        batch = []
        indices = []
        priorities = []
        segment = self.tree.total() / batch_size

        for i in range(batch_size):
            s = random.uniform(segment * i, segment * (i + 1))
            idx, priority, data = self.tree.get(s)
            batch.append(data)
            indices.append(idx)
            priorities.append(priority)

        probs = np.array(priorities) / self.tree.total()
        weights = (len(self.tree) * probs) ** (-beta)
        weights /= weights.max()
        return batch, weights, indices

    def update_priorities(self, indices: List[int], priorities: List[float]) -> None:
        """Update priorities for given indices."""
        for idx, priority in zip(indices, priorities):
            self.tree.update(idx, priority)
            self.max_priority = max(self.max_priority, priority)

    def __len__(self) -> int:
        return len(self.tree)

class MockFPGA:
    """Mock FPGA simulator for DQN inference with latency and quantization."""
    def __init__(self, latency_min: float = CONFIG['fpga_latency_min'], latency_max: float = CONFIG['fpga_latency_max']):
        self.latency_min = latency_min
        self.latency_max = latency_max
        self.bit_precision = 8  # 8-bit fixed-point arithmetic

    def simulate_inference(self, state: torch.Tensor, model: nn.Module) -> torch.Tensor:
        """Simulate FPGA inference with latency and quantization."""
        # Simulate 8-bit quantization
        with torch.no_grad():
            quantized_state = torch.round(state * (2 ** self.bit_precision)) / (2 ** self.bit_precision)
            q_values = model(quantized_state)
            q_values = torch.round(q_values * (2 ** self.bit_precision)) / (2 ** self.bit_precision)

        # Simulate latency
        latency = random.uniform(self.latency_min, self.latency_max)
        time.sleep(latency)
        return q_values

class SubMiniDQN(nn.Module):
    """Single lightweight DQN sub-network for parallel processing."""
    def __init__(self, state_dim: int = CONFIG['state_dim'], action_dim: int = CONFIG['action_dim']):
        super(SubMiniDQN, self).__init__()
        self.fc1 = nn.Linear(state_dim, 12)
        self.fc2 = nn.Linear(12, 6)
        self.fc3 = nn.Linear(6, action_dim)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        return self.fc3(x)

class ParallelMiniDQN(nn.Module):
    """Parallel DQN with multiple sub-networks for ensemble predictions."""
    def __init__(self, state_dim: int = CONFIG['state_dim'], action_dim: int = CONFIG['action_dim'], 
                 num_sub_dqns: int = CONFIG['num_parallel_dqns']):
        super(ParallelMiniDQN, self).__init__()
        self.sub_dqns = nn.ModuleList([SubMiniDQN(state_dim, action_dim) for _ in range(num_sub_dqns)])
        self.num_sub_dqns = num_sub_dqns

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """Aggregate Q-values from all sub-DQNs via averaging."""
        q_values = [sub_dqn(x) for sub_dqn in self.sub_dqns]
        return torch.stack(q_values).mean(dim=0)

class InternodesConsensus:
    """Consensus mechanism for aggregating parallel DQN decisions."""
    def __init__(self, threshold: float = CONFIG['consensus_threshold']):
        self.threshold = threshold

    def aggregate(self, q_values_list: List[torch.Tensor], state: torch.Tensor, 
                  sub_dqns: nn.ModuleList) -> int:
        """Select action via weighted voting based on Q-value confidence."""
        actions = []
        confidences = []
        for i, sub_dqn in enumerate(sub_dqns):
            with torch.no_grad():
                q_vals = sub_dqn(state)
                action = torch.argmax(q_vals).item()
                confidence = torch.softmax(q_vals, dim=0)[action].item()
                actions.append(action)
                confidences.append(confidence)

        # Weighted voting
        vote_counts = np.zeros(CONFIG['action_dim'])
        for action, confidence in zip(actions, confidences):
            if confidence >= self.threshold:
                vote_counts[action] += confidence
            else:
                vote_counts[action] += confidence * 0.5  # Discount low-confidence votes

        return int(np.argmax(vote_counts))

class TinyLSTM(nn.Module):
    """Tiny LSTM for syndrome prediction, optimized for low power."""
    def __init__(self, input_dim: int = CONFIG['state_dim'], hidden_dim: int = 8):
        super(TinyLSTM, self).__init__()
        self.lstm = nn.LSTM(input_dim, hidden_dim, 1, batch_first=True, bidirectional=CONFIG['bidirectional_lstm'])
        self.fc = nn.Linear(hidden_dim * (2 if CONFIG['bidirectional_lstm'] else 1), input_dim)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        _, (h_n, _) = self.lstm(x)
        return torch.sigmoid(self.fc(h_n[-1]))

class QuantumBackend:
    """Quantum backend for simulation or hardware integration."""
    def __init__(self, backend_type: str = 'simulator', hardware_name: Optional[str] = None):
        self.backend_type = backend_type
        self.backend = None
        self._initialize_backend(hardware_name)

    def _initialize_backend(self, hardware_name: Optional[str]) -> None:
        """Initialize backend with retry logic for hardware."""
        if self.backend_type == 'simulator':
            self.backend = 'classical_simulator'
            logging.info("Initialized classical simulator backend.")
            return

        if self.backend_type != 'hardware':
            logging.warning("Unknown backend type. Using simulator.")
            self.backend = 'classical_simulator'
            return

        try:
            from qiskit import QuantumCircuit, execute, IBMQ
            max_retries = 3
            for attempt in range(max_retries):
                try:
                    IBMQ.load_account()
                    provider = IBMQ.get_provider()
                    self.backend = provider.get_backend(hardware_name or 'ibmq_qasm_simulator')
                    self.calibration = {'T1': 100, 'T2': 200, 'gate_time': 50}
                    logging.info(f"Initialized Qiskit backend: {self.backend}")
                    return
                except Exception as e:
                    if attempt == max_retries - 1:
                        logging.error(f"Hardware backend failed after {max_retries} attempts: {e}. Using simulator.")
                        self.backend = 'classical_simulator'
                    else:
                        time.sleep(2 ** attempt)
        except ImportError:
            logging.error("Qiskit not installed. Using simulator.")
            self.backend = 'classical_simulator'

class QuantumPhoneEnv:
    """Environment for 5-qubit repetition code with phone-specific noise."""
    def __init__(self, use_hardware: bool = False):
        self.num_qubits = 5
        self.num_ancilla = 4
        self.backend = QuantumBackend(backend_type='hardware' if use_hardware else 'simulator')
        self.noise_params = CONFIG['noise_params']
        self.noise_covariance = CONFIG['noise_covariance']
        self.max_steps = CONFIG['max_steps']
        self.current_step = 0
        self.state = None
        self.noise_history = deque(maxlen=10)

    def _get_phone_temp(self) -> float:
        """Simulate phone temperature (°C) with Gaussian distribution."""
        return max(25, np.random.normal(loc=35, scale=5))

    def _get_rf_level(self) -> float:
        """Simulate RF interference level (0 to 1) with Beta distribution."""
        return np.random.beta(a=2, b=5)

    def _get_emi_level(self) -> float:
        """Simulate EMI level (0 to 1) with Beta distribution."""
        return np.random.beta(a=3, b=4)

    def _compute_total_noise(self) -> float:
        """Compute total noise probability with correlated sources."""
        temp = self._get_phone_temp()
        temp_error = self.noise_params['phone_temp']['error_per_degree'] * max(0, temp - self.noise_params['phone_temp']['base_temp'])
        rf_level = self._get_rf_level()
        emi_level = self._get_emi_level()
        rf_error = self.noise_params['phone_rf']['error_rate'] * rf_level
        emi_error = self.noise_params['phone_emi']['error_rate'] * emi_level
        thermal_error = (1 - np.exp(-self.noise_params['thermal']['gate_time'] / self.noise_params['thermal']['t1'])) + \
                        (1 - np.exp(-self.noise_params['thermal']['gate_time'] / self.noise_params['thermal']['t2'])) / 2
        base_error = self.noise_params['depolarizing']['error_rate']

        cov_rf_emi = self.noise_covariance['rf_emi'] * np.sqrt(rf_error * emi_error)
        noise_vector = np.array([base_error, temp_error, rf_error, emi_error, thermal_error])
        cov_matrix = np.diag(np.ones(len(noise_vector))) * 0.01
        cov_matrix[2, 3] = cov_matrix[3, 2] = cov_rf_emi
        correlated_noise = np.sqrt(np.sum(np.random.multivariate_normal(noise_vector, cov_matrix) ** 2))
        total_noise = min(0.5, correlated_noise)
        self.noise_history.append(total_noise)
        return total_noise

    def validate_syndrome(self, syndrome: str) -> str:
        """Validate that syndrome is a 4-bit binary string."""
        if not (isinstance(syndrome, str) and len(syndrome) == 4 and set(syndrome).issubset({'0', '1'})):
            logging.error(f"Invalid syndrome: {syndrome}. Generating random syndrome.")
            return ''.join(random.choice('01') for _ in range(4))
        return syndrome

    def reset(self) -> str:
        """Reset environment to a random error state."""
        self.current_step = 0
        self.state = random.randint(0, 31)
        return self.get_syndrome()

    def get_syndrome(self, batch: Optional[List[int]] = None) -> Union[str, List[str]]:
        """Simulate syndrome measurement for 5-qubit repetition code, supporting batch processing."""
        if batch is not None:
            return [self._get_single_syndrome(state) for state in batch]
        return self._get_single_syndrome(self.state)

    def _get_single_syndrome(self, state: int) -> str:
        """Helper for single syndrome generation."""
        try:
            error_state = bin(state)[2:].zfill(5)
            syndrome = ''
            noise_prob = self._compute_total_noise()
            for i in range(4):
                s = '1' if error_state[i] != error_state[i+1] else '0'
                s = '1' if s == '0' and random.random() < noise_prob else '0' if s == '1' and random.random() < noise_prob else s
                syndrome += s
            return self.validate_syndrome(syndrome)
        except Exception as e:
            logging.error(f"Syndrome generation failed: {e}. Returning random syndrome.")
            return ''.join(random.choice('01') for _ in range(4))

    def step(self, action: int) -> Tuple[str, float, bool]:
        """Apply action and return next syndrome, reward, and done flag."""
        if not 0 <= action < CONFIG['action_dim']:
            raise ValueError(f"Invalid action: {action}. Must be in [0, {CONFIG['action_dim']-1}]")
        prev_errors = bin(self.state).count('1')
        if action < 5:
            self.state ^= (1 << action)
        new_syndrome = self.get_syndrome()
        errors = bin(self.state).count('1')
        reward = 1.0 if new_syndrome == '0000' else (0.5 * (prev_errors - errors) / prev_errors if prev_errors > 0 else -0.1)
        self.current_step += 1
        done = new_syndrome == '0000' or self.current_step >= self.max_steps
        return new_syndrome, reward, done

class MiniQECAgent:
    """MiniQEC agent with ParallelMiniDQN and TinyLSTM for phone-compatible QEC."""
    def __init__(self, state_dim: int = CONFIG['state_dim'], action_dim: int = CONFIG['action_dim']):
        self.device = torch.device('cpu')
        self.dqn = ParallelMiniDQN(state_dim, action_dim).to(self.device)
        self.target_dqn = ParallelMiniDQN(state_dim, action_dim).to(self.device)
        self.target_dqn.load_state_dict(self.dqn.state_dict())
        hidden_dim = 4 if self._is_low_resource_device() else 8
        self.lstm = TinyLSTM(hidden_dim=hidden_dim).to(self.device)
        self.dqn = torch.quantization.quantize_dynamic(self.dqn, {nn.Linear}, dtype=torch.qint8)
        self.target_dqn = torch.quantization.quantize_dynamic(self.target_dqn, {nn.Linear}, dtype=torch.qint8)
        self.lstm = torch.quantization.quantize_dynamic(self.lstm, {nn.LSTM, nn.Linear}, dtype=torch.qint8)
        self.optimizer = torch.optim.Adam(self.dqn.parameters(), lr=CONFIG['lr_dqn'])
        self.lstm_optimizer = torch.optim.Adam(self.lstm.parameters(), lr=CONFIG['lr_lstm'])
        self.scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(self.optimizer, mode='max', factor=0.5, patience=20)
        self.lstm_scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(self.lstm_optimizer, mode='min', factor=0.5, patience=20)
        self.memory = PriorityReplayBuffer(CONFIG['memory_capacity'])
        self.batch_size = CONFIG['batch_size']
        self.gamma = CONFIG['gamma']
        self.epsilon = CONFIG['epsilon']
        self.epsilon_min = CONFIG['epsilon_min']
        self.epsilon_decay = CONFIG['epsilon_decay']
        self.per_alpha = CONFIG['per_alpha']
        self.per_beta = CONFIG['per_beta']
        self.per_beta_increment = CONFIG['per_beta_increment']
        self.alpha_lstm = CONFIG['alpha_lstm_base']
        self.sequence_length = CONFIG['sequence_length_base']
        self.syndrome_history = deque(maxlen=self.sequence_length)
        self.step_counter = 0
        self.huber_loss = nn.HuberLoss()
        self.lstm_confidence = 0.5
        self.fpga_enabled = CONFIG['fpga_enabled']
        self.consensus = InternodesConsensus()
        self.mock_fpga = MockFPGA()
        # Fixed: store episode rewards for hyperparameter tuning compatibility
        self.episode_rewards: List[float] = []

    def _is_low_resource_device(self) -> bool:
        """Check if device has limited resources (e.g., low memory)."""
        try:
            mem = psutil.virtual_memory()
            return mem.available < 2 * 1024 * 1024 * 1024
        except:
            return False

    def _compute_td_error(self, state: str, action: int, reward: float, next_state: str, done: bool) -> float:
        """Compute TD error for prioritized experience replay."""
        state_tensor = torch.tensor([float(s) for s in state], dtype=torch.float32).to(self.device)
        next_state_tensor = torch.tensor([float(s) for s in next_state], dtype=torch.float32).to(self.device)
        q_value = self.dqn(state_tensor)[action]
        next_action = self.dqn(next_state_tensor).argmax().item()
        next_q_value = self.target_dqn(next_state_tensor)[next_action].item()
        target = reward + self.gamma * next_q_value * (1 - done)
        return abs(q_value.item() - target)

    def choose_action(self, state: str, use_lstm: bool = True, episode: int = 0, 
                     total_episodes: int = CONFIG['episodes']) -> int:
        """Select action using epsilon-greedy policy with LSTM prediction blend OR internodes consensus.
        BUG FIXED: LSTM-blended Q-values are now actually used for action selection when LSTM is active.
        Epsilon decay moved to train loop (per-episode) for correct RL behavior."""
        # Update LSTM blending weight (per-episode basis)
        self.alpha_lstm = min(0.5, CONFIG['alpha_lstm_base'] + 0.3 * (episode / total_episodes))

        if random.random() < self.epsilon:
            return random.randint(0, CONFIG['action_dim'] - 1)

        state_tensor = torch.tensor([float(s) for s in state], dtype=torch.float32).to(self.device)
        if self.fpga_enabled:
            return self._fpga_inference(state_tensor)

        with torch.no_grad():
            if use_lstm and self.syndrome_history and random.random() < min(0.9, self.lstm_confidence):
                lstm_input = torch.tensor([[float(s) for s in h] for h in self.syndrome_history], 
                                        dtype=torch.float32).unsqueeze(0).to(self.device)
                pred_syndrome = self.lstm(lstm_input)
                current_q = self.dqn(state_tensor)
                pred_q_values = self.dqn(pred_syndrome)
                q_values = (1 - self.alpha_lstm) * current_q + self.alpha_lstm * pred_q_values
                return torch.argmax(q_values).item()
            else:
                # Robust ensemble decision via internodes consensus when LSTM prediction is not used
                return self.consensus.aggregate(
                    [sub_dqn(state_tensor) for sub_dqn in self.dqn.sub_dqns], 
                    state_tensor, self.dqn.sub_dqns
                )

    def _fpga_inference(self, state: torch.Tensor) -> int:
        """Perform inference using mock FPGA."""
        q_values = self.mock_fpga.simulate_inference(state, self.dqn)
        return torch.argmax(q_values).item()

    def train_dqn(self, state: str, action: int, reward: float, next_state: str, done: bool) -> None:
        """Train ParallelMiniDQN with Double DQN and prioritized experience replay."""
        priority = self._compute_td_error(state, action, reward, next_state, done) + 1e-5
        self.memory.append(priority, (state, action, reward, next_state, done))
        if len(self.memory) < max(self.batch_size, CONFIG['warmup_steps']):
            return

        batch, weights, indices = self.memory.sample(self.batch_size, self.per_alpha, self.per_beta)
        states = torch.tensor([[float(s) for s in t[0]] for t in batch], dtype=torch.float32).to(self.device)
        actions = torch.tensor([t[1] for t in batch], dtype=torch.long).to(self.device)
        rewards = torch.tensor([t[2] for t in batch], dtype=torch.float32).to(self.device)
        next_states = torch.tensor([[float(s) for s in t[3]] for t in batch], dtype=torch.float32).to(self.device)
        dones = torch.tensor([t[4] for t in batch], dtype=torch.float32).to(self.device)

        q_values = self.dqn(states).gather(1, actions.unsqueeze(1)).squeeze(1)
        with torch.no_grad():
            next_actions = self.dqn(next_states).argmax(1)
            next_q_values = self.target_dqn(next_states).gather(1, next_actions.unsqueeze(1)).squeeze(1)
            targets = rewards + self.gamma * next_q_values * (1 - dones)

        loss = self.huber_loss(q_values, targets) * torch.tensor(weights, dtype=torch.float32).to(self.device).mean()
        self.optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(self.dqn.parameters(), max_norm=1.0)
        self.optimizer.step()

        td_errors = torch.abs(q_values - targets).detach().cpu().numpy()
        self.memory.update_priorities(indices, td_errors + 1e-5)
        self.per_beta = min(1.0, self.per_beta + self.per_beta_increment)
        self.step_counter += 1
        if self.step_counter % 50 == 0:
            self.target_dqn.load_state_dict(self.dqn.state_dict())

    def train_lstm(self, syndrome: str, next_syndrome: str) -> None:
        """Train LSTM to predict next syndrome with log-likelihood confidence."""
        self.syndrome_history.append(syndrome)
        if len(self.syndrome_history) < self.sequence_length:
            return

        lstm_input = torch.tensor([[float(s) for s in h] for h in self.syndrome_history], 
                                dtype=torch.float32).unsqueeze(0).to(self.device)
        target = torch.tensor([float(s) for s in next_syndrome], dtype=torch.float32).unsqueeze(0).to(self.device)
        pred = torch.clamp(self.lstm(lstm_input), 1e-7, 1 - 1e-7)
        loss = nn.BCELoss()(pred, target)
        self.lstm_optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(self.lstm.parameters(), max_norm=1.0)
        self.lstm_optimizer.step()

        log_likelihood = -nn.BCELoss(reduction='none')(pred, target).mean().item()
        self.lstm_confidence = 1 / (1 + np.exp(-log_likelihood))

def monitor_system_resources(threshold_battery: float = 20.0, threshold_cpu: float = 80.0) -> bool:
    """Check battery and CPU usage to determine if training should pause."""
    try:
        battery = psutil.sensors_battery()
        cpu_percent = psutil.cpu_percent(interval=1)
        if battery and battery.percent < threshold_battery:
            logging.warning(f"Battery low: {battery.percent}%. Pausing training.")
            return True
        if cpu_percent > threshold_cpu:
            logging.warning(f"CPU usage high: {cpu_percent}%. Pausing training.")
            return True
        return False
    except:
        logging.warning("Resource monitoring unavailable. Assuming sufficient resources.")
        return False

def save_encrypted_model(model: nn.Module, filename: str, key: bytes) -> None:
    """Save model weights encrypted for security."""
    cipher = Fernet(key)
    with tempfile.NamedTemporaryFile(delete=False) as temp_file:
        torch.save(model.state_dict(), temp_file.name)
        with open(temp_file.name, 'rb') as f:
            encrypted_data = cipher.encrypt(f.read())
        with open(filename, 'wb') as f:
            f.write(encrypted_data)
        os.unlink(temp_file.name)

def load_encrypted_model(model: nn.Module, filename: str, key: bytes) -> None:
    """Load encrypted model weights."""
    cipher = Fernet(key)
    with open(filename, 'rb') as f:
        encrypted_data = f.read()
    with tempfile.NamedTemporaryFile(delete=False) as temp_file:
        temp_file.write(cipher.decrypt(encrypted_data))
        temp_file.flush()
        model.load_state_dict(torch.load(temp_file.name))
        os.unlink(temp_file.name)

def train_miniqec(episodes: int = CONFIG['episodes'], use_hardware: bool = False, 
                 pretrained_path: Optional[str] = None) -> Optional['MiniQECAgent']:
    """Train MiniQEC agent with early stopping, metrics logging, and encryption.
    BUG FIXED: episode_rewards now stored on agent for hyperparameter tuning."""
    if monitor_system_resources():
        logging.info("Insufficient resources. Pausing training.")
        return None

    env = QuantumPhoneEnv(use_hardware)
    agent = MiniQECAgent()
    encryption_key = Fernet.generate_key()
    if pretrained_path and os.path.exists(pretrained_path):
        load_encrypted_model(agent.dqn, pretrained_path, encryption_key)
        agent.target_dqn.load_state_dict(agent.dqn.state_dict())
        logging.info(f"Loaded pretrained model from {pretrained_path}")

    rewards, losses, successes = [], [], []
    best_reward = -float('inf')
    patience_counter = 0
    metrics_file = 'training_metrics.csv'
    metrics_json = 'training_metrics.json'
    profile_enabled = False

    with open(metrics_file, 'w', newline='') as f:
        writer = csv.writer(f)
        writer.writerow(['Episode', 'Avg_Reward', 'Avg_Loss', 'Success_Rate'])
    with open(metrics_json, 'w') as f:
        json.dump([], f)

    for episode in range(episodes):
        if monitor_system_resources():
            logging.info("Resource constraints detected. Saving checkpoint and pausing.")
            save_encrypted_model(agent.dqn, f'miniqec_dqn_checkpoint_{episode}.pth', encryption_key)
            return agent

        noise_level = env._compute_total_noise()
        agent.sequence_length = min(CONFIG['sequence_length_base'] + int(noise_level * 5), 10)
        agent.syndrome_history = deque(maxlen=agent.sequence_length)
        state = env.reset()
        total_reward = 0
        episode_losses = []
        done = False
        steps = 0

        profile_enabled = (episode % 100 == 0)
        profiler = torch.profiler.profile(record_shapes=True) if profile_enabled else None
        if profile_enabled:
            profiler.__enter__()

        while not done:
            action = agent.choose_action(state, episode=episode, total_episodes=episodes)
            next_state, reward, done = env.step(action)
            agent.train_dqn(state, action, reward, next_state, done)
            agent.train_lstm(state, next_state)
            total_reward += reward
            episode_losses.append(agent._compute_td_error(state, action, reward, next_state, done))
            state = next_state
            steps += 1

        if profile_enabled:
            profiler.__exit__(None, None, None)
            with open(f'profiling_results_{episode}.txt', 'w') as f:
                f.write(profiler.key_averages().table(sort_by="cpu_time_total"))

        # Store on agent for tuning compatibility
        agent.episode_rewards.append(total_reward)
        rewards.append(total_reward)
        losses.append(np.mean(episode_losses) if episode_losses else 0)
        successes.append(1 if total_reward >= 0.5 else 0)
        agent.scheduler.step(total_reward)
        agent.lstm_scheduler.step(losses[-1])

        # BUG FIXED: epsilon decay is now per-episode (standard RL) instead of per-step inside choose_action
        agent.epsilon = max(agent.epsilon_min, agent.epsilon * agent.epsilon_decay)

        if episode % 50 == 0:
            avg_reward = np.mean(rewards[-50:]) if rewards else 0
            avg_loss = np.mean(losses[-50:]) if losses else 0
            success_rate = np.mean(successes[-50:]) * 100 if successes else 0
            logging.info(f"Episode {episode}, Avg Reward: {avg_reward:.2f}, Avg Loss: {avg_loss:.4f}, Success Rate: {success_rate:.2f}%")

            with open(metrics_file, 'a', newline='') as f:
                writer = csv.writer(f)
                writer.writerow([episode, avg_reward, avg_loss, success_rate])

            with open(metrics_json, 'r') as f:
                metrics_data = json.load(f)
            metrics_data.append({
                'episode': episode,
                'avg_reward': avg_reward,
                'avg_loss': avg_loss,
                'success_rate': success_rate
            })
            with open(metrics_json, 'w') as f:
                json.dump(metrics_data, f, indent=2)

            if avg_reward > best_reward or episode % 50 == 0:
                best_reward = max(best_reward, avg_reward)
                filename = f'miniqec_dqn_best.pth' if avg_reward > best_reward else f'miniqec_dqn_{episode}.pth'
                save_encrypted_model(agent.dqn, filename, encryption_key)
                logging.info(f"Saved model to {filename}")

            if avg_reward > best_reward:
                patience_counter = 0
            else:
                patience_counter += 1
            if patience_counter >= CONFIG['early_stop_patience'] and avg_reward >= CONFIG['early_stop_min_reward']:
                logging.info(f"Early stopping at episode {episode} with avg reward {avg_reward:.2f}")
                break

    return agent

def test_miniqec(agent: 'MiniQECAgent', trials: int = CONFIG['test_trials'], use_hardware: bool = False, 
                use_lstm: bool = True) -> Dict:
    """Test MiniQEC agent across dynamic noise levels and edge cases."""
    env = QuantumPhoneEnv(use_hardware)
    results = {}
    test_conditions = [
        {'name': 'low_noise', 'depolarizing': 0.01, 'temp': 25, 'rf': 0.1, 'emi': 0.1},
        {'name': 'med_noise', 'depolarizing': 0.05, 'temp': 35, 'rf': 0.5, 'emi': 0.5},
        {'name': 'high_noise', 'depolarizing': 0.1, 'temp': 45, 'rf': 1.0, 'emi': 1.0},
        {'name': 'dynamic_noise', 'depolarizing': None, 'temp': None, 'rf': None, 'emi': None},
        {'name': 'adversarial_noise', 'depolarizing': 0.5, 'temp': 50, 'rf': 1.0, 'emi': 1.0}
    ]

    for condition in test_conditions:
        successes = 0
        steps_list = []
        success_rates = []

        for _ in range(trials):
            if condition['depolarizing'] is not None:
                env.noise_params['depolarizing']['error_rate'] = condition['depolarizing']
                env.noise_params['phone_temp']['base_temp'] = condition['temp']
                env.noise_params['phone_rf']['error_rate'] = condition['rf'] * 0.02
                env.noise_params['phone_emi']['error_rate'] = condition['emi'] * 0.015
            else:
                env.noise_params['depolarizing']['error_rate'] = np.random.uniform(0.01, 0.1)
                env.noise_params['phone_temp']['base_temp'] = np.random.uniform(25, 45)
                if len(env.noise_history) > 1:
                    env.noise_params['depolarizing']['error_rate'] *= random.uniform(0.8, 1.2)

            state = env.reset()
            done = False
            steps = 0
            while not done:
                action = agent.choose_action(state, use_lstm=use_lstm)
                state, _, done = env.step(action)
                steps += 1
                if state == '0000':
                    successes += 1
                    steps_list.append(steps)
            success_rates.append(1 if state == '0000' else 0)

        success_rate = successes / trials * 100
        ler = 1 - success_rate / 100
        avg_steps = np.mean(steps_list) if steps_list else 0
        bootstrap_samples = [np.mean(np.random.choice(success_rates, size=trials, replace=True)) for _ in range(1000)]
        ci_lower, ci_upper = np.percentile(bootstrap_samples, [2.5, 97.5]) * 100
        logging.info(f"{condition['name']}: Success Rate: {success_rate:.2f}% [{ci_lower:.2f}-{ci_upper:.2f}], "
                     f"LER: {ler:.4f}, Avg Steps: {avg_steps:.2f}")
        results[condition['name']] = {
            'success_rate': success_rate,
            'ler': ler,
            'avg_steps': avg_steps,
            'ci_lower': ci_lower,
            'ci_upper': ci_upper
        }

    return results

def run_unit_tests() -> None:
    """Run unit tests for critical components.
    BUGS FIXED: syndrome test now validates format only (noise is stochastic),
    do-nothing reward corrected to 0.0, invalid action now raises ValueError."""
    logging.info("Running unit tests...")
    env = QuantumPhoneEnv()
    agent = MiniQECAgent()

    # Test syndrome generation
    env.state = 0b00101
    env.noise_params['depolarizing']['error_rate'] = 0
    # Other noise sources remain; test only validity (deterministic exact string was impossible)
    syndrome = env.get_syndrome()
    assert len(syndrome) == 4 and set(syndrome).issubset({'0', '1'}), f"Syndrome test failed: got {syndrome}"

    # Test reward function
    env.state = 0b00101
    _, reward, _ = env.step(5)
    assert reward == 0.0, f"Reward test failed for do-nothing: got {reward}, expected 0.0 (no change in errors)"
    env.state = 0b00101
    _, reward, _ = env.step(2)
    assert reward > 0, f"Reward test failed: got {reward}, expected > 0"

    # Test parallel DQN
    state_tensor = torch.tensor([float(s) for s in '1010'], dtype=torch.float32).to(agent.device)
    q_values = agent.dqn(state_tensor)
    assert q_values.shape == (CONFIG['action_dim'],), f"Parallel DQN output shape test failed: got {q_values.shape}"

    # Test consensus
    actions = agent.consensus.aggregate([sub_dqn(state_tensor) for sub_dqn in agent.dqn.sub_dqns], 
                                       state_tensor, agent.dqn.sub_dqns)
    assert 0 <= actions < CONFIG['action_dim'], f"Consensus test failed: got action {actions}"

    # Test FPGA inference
    action = agent._fpga_inference(state_tensor)
    assert 0 <= action < CONFIG['action_dim'], f"FPGA inference test failed: got action {action}"

    # Test edge cases
    env.noise_params['depolarizing']['error_rate'] = 0.5
    syndrome = env.get_syndrome()
    assert len(syndrome) == 4 and set(syndrome).issubset({'0', '1'}), f"Max noise syndrome test failed: {syndrome}"
    try:
        env.step(6)
        assert False, "Invalid action test failed: No exception raised"
    except ValueError:
        pass  # Expected

    logging.info("All unit tests passed!")

def hyperparameter_tuning(episodes: int = 100) -> Optional[Dict]:
    """Hyperparameter tuning with Optuna.
    BUG FIXED: objective now correctly reads agent.episode_rewards (added to MiniQECAgent)."""
    try:
        ray.init(ignore_reinit_error=True)

        @ray.remote
        def objective(trial):
            config = CONFIG.copy()
            config['lr_dqn'] = trial.suggest_float('lr_dqn', 1e-4, 1e-2, log=True)
            config['lr_lstm'] = trial.suggest_float('lr_lstm', 1e-4, 1e-2, log=True)
            config['gamma'] = trial.suggest_float('gamma', 0.9, 0.999)
            config['per_alpha'] = trial.suggest_float('per_alpha', 0.4, 0.8)
            config['per_beta'] = trial.suggest_float('per_beta', 0.3, 0.7)
            config['sequence_length_base'] = trial.suggest_int('sequence_length_base', 2, 5)
            # Apply config for this trial (global works reliably in local Ray)
            global CONFIG
            CONFIG = config
            agent = train_miniqec(episodes=episodes)
            if agent is None or not agent.episode_rewards:
                return -float('inf')
            return np.mean(agent.episode_rewards[-10:])

        study = optuna.create_study(direction='maximize')
        study.optimize(lambda trial: ray.get(objective.remote(trial)), n_trials=50, n_jobs=2)
        logging.info(f"Best hyperparameters: {study.best_params}")
        ray.shutdown()
        return study.best_params
    except ImportError:
        logging.warning("Optuna or Ray not installed. Skipping hyperparameter tuning.")
        return None
