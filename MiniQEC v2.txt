@dataclass
class MiniQECConfig:
    """Main configuration container with validation."""
    noise: NoiseConfig = field(default_factory=NoiseConfig)
    rl: RLConfig = field(default_factory=RLConfig)
    hardware: HardwareConfig = field(default_factory=HardwareConfig)
    seed: int = 42
    log_level: str = "INFO"
    enable_profiling: bool = False
    curriculum_learning: bool = True
    
    def __post_init__(self):
        """Validate configuration after initialization."""
        self._validate()
    
    def _validate(self):
        """Validate all configuration values."""
        assert 0 <= self.rl.gamma <= 1, f"Gamma must be in [0, 1], got {self.rl.gamma}"
        assert self.rl.batch_size > 0, f"Batch size must be positive"
        assert self.rl.memory_capacity >= self.rl.batch_size, "Memory capacity must be >= batch size"
        assert self.rl.state_dim > 0, f"State dim must be positive"
        assert self.rl.action_dim > 0, f"Action dim must be positive"
        assert self.rl.num_parallel_dqns >= 1, f"Number of parallel DQNs must be >= 1"
        assert 0 < self.rl.consensus_threshold <= 1, f"Consensus threshold must be in (0, 1]"
        assert self.hardware.fpga_latency_min >= 0, "FPGA latency must be non-negative"
        assert self.hardware.fpga_latency_max >= self.hardware.fpga_latency_min, "Max latency must be >= min"
    
    def to_dict(self) -> Dict:
        """Convert to dictionary for serialization."""
        return asdict(self)
    
    @classmethod
    def from_dict(cls, data: Dict) -> 'MiniQECConfig':
        """Create config from dictionary."""
        noise = NoiseConfig(**data.get('noise', {}))
        rl = RLConfig(**data.get('rl', {}))
        hardware = HardwareConfig(**data.get('hardware', {}))
        return cls(
            noise=noise,
            rl=rl,
            hardware=hardware,
            seed=data.get('seed', 42),
            log_level=data.get('log_level', 'INFO'),
            enable_profiling=data.get('enable_profiling', False),
            curriculum_learning=data.get('curriculum_learning', True)
        )

class StructuredLogger:
    """Enterprise-grade structured logging with JSON support."""
    
    def __init__(self, name: str = "MiniQEC", log_file: str = "miniqec.log", 
                 level: str = "INFO", enable_json: bool = True):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(getattr(logging, level.upper()))
        self.enable_json = enable_json
        
        # Clear existing handlers
        self.logger.handlers.clear()
        
        # File handler with JSON formatting
        file_handler = logging.FileHandler(log_file)
        file_handler.setFormatter(self._get_formatter(enable_json))
        self.logger.addHandler(file_handler)
        
        # Console handler
        console_handler = logging.StreamHandler()
        console_handler.setFormatter(self._get_formatter(False))
        self.logger.addHandler(console_handler)
    
    def _get_formatter(self, json_format: bool) -> logging.Formatter:
        """Get appropriate formatter."""
        if json_format:
            return logging.Formatter(
                '{"timestamp": "%(asctime)s", "level": "%(levelname)s", '
                '"message": "%(message)s", "module": "%(name)s"}'
            )
        return logging.Formatter('%(asctime)s - %(levelname)s - %(message)s')
    
    def info(self, msg: str, **kwargs):
        self.logger.info(msg, extra=kwargs)
    
    def warning(self, msg: str, **kwargs):
        self.logger.warning(msg, extra=kwargs)
    
    def error(self, msg: str, **kwargs):
        self.logger.error(msg, extra=kwargs)
    
    def debug(self, msg: str, **kwargs):
        self.logger.debug(msg, extra=kwargs)

# Global logger instance

def set_seed(seed: int = 42) -> None:
    """Set random seeds for reproducibility."""
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    if torch.cuda.is_available():
        torch.cuda.manual_seed_all(seed)
    logger.info(f"Random seed set to {seed}")

def compute_model_hash(filepath: str) -> str:
    """Compute SHA256 hash of model file for integrity verification."""
    sha256_hash = hashlib.sha256()
    with open(filepath, "rb") as f:
        for byte_block in iter(lambda: f.read(4096), b""):
            sha256_hash.update(byte_block)
    return sha256_hash.hexdigest()

def verify_model_integrity(filepath: str, expected_hash: str) -> bool:
    """Verify model file integrity using hash comparison."""
    actual_hash = compute_model_hash(filepath)
    if actual_hash != expected_hash:
        logger.error(f"Model integrity check failed. Expected: {expected_hash}, Got: {actual_hash}")
        return False
    logger.info("Model integrity verification passed")
    return True

class HardwareAccelerator(ABC):
    """Abstract base class for hardware accelerators."""
    
    @abstractmethod
    async def infer(self, state: torch.Tensor) -> torch.Tensor:
        """Perform inference asynchronously."""
        pass
    
    @abstractmethod
    def get_latency_stats(self) -> Dict[str, float]:
        """Get latency statistics."""
        pass

class MockFPGA(HardwareAccelerator):
    """Mock FPGA simulator with async support and quantization."""
    
    def __init__(self, config: HardwareConfig = DEFAULT_CONFIG.hardware):
        self.latency_min = config.fpga_latency_min
        self.latency_max = config.fpga_latency_max
        self.bit_precision = config.bit_precision
        self.latency_history: List[float] = []
    
    async def infer(self, state: torch.Tensor, model: nn.Module) -> torch.Tensor:
        """Simulate FPGA inference with async latency and quantization."""
        start_time = time.perf_counter()
        
        # Simulate 8-bit quantization
        with torch.no_grad():
            quantized_state = torch.round(state * (2 ** self.bit_precision)) / (2 ** self.bit_precision)
            q_values = model(quantized_state)
            q_values = torch.round(q_values * (2 ** self.bit_precision)) / (2 ** self.bit_precision)
        
        # Simulate async latency
        latency = random.uniform(self.latency_min, self.latency_max)
        await asyncio.sleep(latency)
        
        end_time = time.perf_counter()
        self.latency_history.append(end_time - start_time)
        
        return q_values
    
    def get_latency_stats(self) -> Dict[str, float]:
        """Get latency statistics from history."""
        if not self.latency_history:
            return {'min': 0, 'max': 0, 'avg': 0, 'std': 0}
        return {
            'min': min(self.latency_history),
            'max': max(self.latency_history),
            'avg': np.mean(self.latency_history),
            'std': np.std(self.latency_history)
        }
    
    def reset_stats(self):
        """Reset latency statistics."""
        self.latency_history.clear()

class CPUAccelerator(HardwareAccelerator):
    """CPU-based inference accelerator."""
    
    def __init__(self):
        self.latency_history: List[float] = []
    
    async def infer(self, state: torch.Tensor, model: nn.Module) -> torch.Tensor:
        """Perform CPU inference."""
        start_time = time.perf_counter()
        with torch.no_grad():
            q_values = model(state)
        end_time = time.perf_counter()
        self.latency_history.append(end_time - start_time)
        return q_values
    
    def get_latency_stats(self) -> Dict[str, float]:
        if not self.latency_history:
            return {'min': 0, 'max': 0, 'avg': 0, 'std': 0}
        return {
            'min': min(self.latency_history),
            'max': max(self.latency_history),
            'avg': np.mean(self.latency_history),
            'std': np.std(self.latency_history)
        }
    
    def reset_stats(self):
        self.latency_history.clear()

class QuantumCode(ABC):
    """Abstract base class for quantum error correction codes."""
    
    @abstractmethod
    def get_syndrome(self, state: int) -> str:
        """Generate syndrome from quantum state."""
        pass
    
    @abstractmethod
    def get_num_qubits(self) -> int:
        """Get number of data qubits."""
        pass
    
    @abstractmethod
    def get_num_ancilla(self) -> int:
        """Get number of ancilla qubits."""
        pass

class RepetitionCode5Qubit(QuantumCode):
    """5-qubit repetition code implementation."""
    
    def __init__(self):
        self.num_qubits = 5
        self.num_ancilla = 4
    
    def get_syndrome(self, state: int) -> str:
        """Generate 4-bit syndrome for 5-qubit repetition code."""
        error_state = bin(state)[2:].zfill(self.num_qubits)
        syndrome = ''
        for i in range(self.num_ancilla):
            s = '1' if error_state[i] != error_state[i+1] else '0'
            syndrome += s
        return syndrome
    
    def get_num_qubits(self) -> int:
        return self.num_qubits
    
    def get_num_ancilla(self) -> int:
        return self.num_ancilla
    
    def get_state_dim(self) -> int:
        """Get syndrome dimension for RL state."""
        return self.num_ancilla
    
    def get_action_dim(self) -> int:
        """Get action dimension (bit flips + do nothing)."""
        return self.num_qubits + 1

class QuantizableLinear(nn.Module):
    """Linear layer wrapper for Quantization Aware Training."""
    
    def __init__(self, in_features: int, out_features: int):
        super().__init__()
        self.linear = nn.Linear(in_features, out_features)
        self.quant = quantization.QuantStub()
        self.dequant = quantization.DeQuantStub()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.quant(x)
        x = self.linear(x)
        x = self.dequant(x)
        return x

class SubMiniDQN(nn.Module):
    """Single lightweight DQN sub-network with QAT support."""
    
    def __init__(self, state_dim: int, action_dim: int, use_qat: bool = True):
        super(SubMiniDQN, self).__init__()
        self.use_qat = use_qat
        
        if use_qat:
            self.fc1 = QuantizableLinear(state_dim, 12)
            self.fc2 = QuantizableLinear(12, 6)
            self.fc3 = QuantizableLinear(6, action_dim)
        else:
            self.fc1 = nn.Linear(state_dim, 12)
            self.fc2 = nn.Linear(12, 6)
            self.fc3 = nn.Linear(6, action_dim)
        
        self.relu = nn.ReLU()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.relu(self.fc1(x))
        x = self.relu(self.fc2(x))
        return self.fc3(x)
    
    def fuse_model(self):
        """Fuse modules for better quantization performance."""
        if self.use_qat:
            quantization.fuse_modules(self, [['fc1', 'relu'], ['fc2', 'relu']], inplace=True)

class ParallelMiniDQN(nn.Module):
    """Parallel DQN with multiple sub-networks for ensemble predictions."""
    
    def __init__(self, state_dim: int, action_dim: int, num_sub_dqns: int = 4, 
                 use_qat: bool = True):
        super(ParallelMiniDQN, self).__init__()
        self.sub_dqns = nn.ModuleList([
            SubMiniDQN(state_dim, action_dim, use_qat) for _ in range(num_sub_dqns)
        ])
        self.num_sub_dqns = num_sub_dqns
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """Aggregate Q-values from all sub-DQNs via averaging."""
        q_values = [sub_dqn(x) for sub_dqn in self.sub_dqns]
        return torch.stack(q_values).mean(dim=0)
    
    def fuse_model(self):
        """Fuse all sub-networks."""
        for sub_dqn in self.sub_dqns:
            sub_dqn.fuse_model()

class TinyLSTM(nn.Module):
    """Tiny LSTM for syndrome prediction with QAT support."""
    
    def __init__(self, input_dim: int, hidden_dim: int = 8, 
                 bidirectional: bool = False, use_qat: bool = True):
        super(TinyLSTM, self).__init__()
        self.use_qat = use_qat
        self.lstm = nn.LSTM(input_dim, hidden_dim, 1, batch_first=True, 
                           bidirectional=bidirectional)
        fc_input_dim = hidden_dim * (2 if bidirectional else 1)
        
        if use_qat:
            self.fc = QuantizableLinear(fc_input_dim, input_dim)
        else:
            self.fc = nn.Linear(fc_input_dim, input_dim)
        
        self.sigmoid = nn.Sigmoid()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        _, (h_n, _) = self.lstm(x)
        return self.sigmoid(self.fc(h_n[-1]))

class InternodesConsensus:
    """Consensus mechanism for aggregating parallel DQN decisions."""
    
    def __init__(self, threshold: float = 0.7):
        self.threshold = threshold
    
    def aggregate(self, q_values_list: List[torch.Tensor], state: torch.Tensor,
                  sub_dqns: nn.ModuleList) -> int:
        """Select action via weighted voting based on Q-value confidence."""
        actions = []
        confidences = []
        
        for i, sub_dqn in enumerate(sub_dqns):
            with torch.no_grad():
                q_vals = sub_dqn(state)
                action = torch.argmax(q_vals).item()
                confidence = torch.softmax(q_vals, dim=0)[action].item()
                actions.append(action)
                confidences.append(confidence)
        
        # Weighted voting
        action_dim = q_values_list[0].shape[0] if q_values_list else 6
        vote_counts = np.zeros(action_dim)
        
        for action, confidence in zip(actions, confidences):
            if confidence >= self.threshold:
                vote_counts[action] += confidence
            else:
                vote_counts[action] += confidence * 0.5
        
        return int(np.argmax(vote_counts))

class SumTree:
    """Sum-tree for efficient prioritized experience replay."""
    
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.tree = np.zeros(2 * capacity - 1)
        self.data = np.zeros(capacity, dtype=object)
        self.write_pointer = 0
        self.n_entries = 0
    
    def _propagate(self, idx: int, change: float) -> None:
        parent = (idx - 1) // 2
        self.tree[parent] += change
        if parent != 0:
            self._propagate(parent, change)
    
    def _retrieve(self, idx: int, s: float) -> int:
        left = 2 * idx + 1
        right = left + 1
        if left >= len(self.tree):
            return idx
        if s <= self.tree[left]:
            return self._retrieve(left, s)
        return self._retrieve(right, s - self.tree[left])
    
    def total(self) -> float:
        return self.tree[0]
    
    def append(self, priority: float, data: Tuple) -> None:
        idx = self.write_pointer + self.capacity - 1
        self.data[self.write_pointer] = data
        self.update(idx, priority)
        self.write_pointer = (self.write_pointer + 1) % self.capacity
        self.n_entries = min(self.n_entries + 1, self.capacity)
    
    def update(self, idx: int, priority: float) -> None:
        change = priority - self.tree[idx]
        self.tree[idx] = priority
        self._propagate(idx, change)
    
    def get(self, s: float) -> Tuple[int, float, Tuple]:
        idx = self._retrieve(0, s)
        data_idx = idx - self.capacity + 1
        return idx, self.tree[idx], self.data[data_idx]
    
    def __len__(self) -> int:
        return self.n_entries

class PriorityReplayBuffer:
    """Prioritized experience replay buffer using sum-tree."""
    
    def __init__(self, capacity: int):
        self.tree = SumTree(capacity)
        self.capacity = capacity
        self.max_priority = 1.0
    
    def append(self, priority: float, experience: Tuple) -> None:
        self.max_priority = max(self.max_priority, priority)
        self.tree.append(priority, experience)
    
    def sample(self, batch_size: int, alpha: float = 0.6, 
               beta: float = 0.4) -> Tuple[List, np.ndarray, List]:
        batch = []
        indices = []
        priorities = []
        segment = self.tree.total() / batch_size
        
        for i in range(batch_size):
            s = random.uniform(segment * i, segment * (i + 1))
            idx, priority, data = self.tree.get(s)
            batch.append(data)
            indices.append(idx)
            priorities.append(priority)
        
        probs = np.array(priorities) / self.tree.total()
        weights = (len(self.tree) * probs) ** (-beta)
        weights /= weights.max()
        
        return batch, weights, indices
    
    def update_priorities(self, indices: List[int], priorities: List[float]) -> None:
        for idx, priority in zip(indices, priorities):
            self.tree.update(idx, priority)
            self.max_priority = max(self.max_priority, priority)
    
    def __len__(self) -> int:
        return len(self.tree)

class QuantumBackend:
    """Quantum backend for simulation or hardware integration."""
    
    def __init__(self, backend_type: str = 'simulator', 
                 hardware_name: Optional[str] = None):
        self.backend_type = backend_type
        self.backend = None
        self.calibration = None
        self._initialize_backend(hardware_name)
    
    def _initialize_backend(self, hardware_name: Optional[str]) -> None:
        if self.backend_type == 'simulator':
            self.backend = 'classical_simulator'
            self.calibration = {'T1': 100, 'T2': 200, 'gate_time': 50}
            logger.info("Initialized classical simulator backend")
            return
        
        if self.backend_type != 'hardware':
            logger.warning("Unknown backend type. Using simulator.")
            self.backend = 'classical_simulator'
            self.calibration = {'T1': 100, 'T2': 200, 'gate_time': 50}
            return
        
        try:
            from qiskit import IBMQ
            max_retries = 3
            for attempt in range(max_retries):
                try:
                    IBMQ.load_account()
                    provider = IBMQ.get_provider()
                    self.backend = provider.get_backend(hardware_name or 'ibmq_qasm_simulator')
                    self.calibration = {'T1': 100, 'T2': 200, 'gate_time': 50}
                    logger.info(f"Initialized Qiskit backend: {self.backend}")
                    return
                except Exception as e:
                    if attempt == max_retries - 1:
                        logger.error(f"Hardware backend failed: {e}. Using simulator.")
                        self.backend = 'classical_simulator'
                        self.calibration = {'T1': 100, 'T2': 200, 'gate_time': 50}
                    else:
                        time.sleep(2 ** attempt)
        except ImportError:
            logger.error("Qiskit not installed. Using simulator.")
            self.backend = 'classical_simulator'
            self.calibration = {'T1': 100, 'T2': 200, 'gate_time': 50}

class QuantumPhoneEnv:
    """Environment for 5-qubit repetition code with phone-specific noise."""
    
    def __init__(self, config: MiniQECConfig = DEFAULT_CONFIG, 
                 use_hardware: bool = False, quantum_code: Optional[QuantumCode] = None):
        self.config = config
        self.quantum_code = quantum_code or RepetitionCode5Qubit()
        self.backend = QuantumBackend(
            backend_type='hardware' if use_hardware else 'simulator'
        )
        self.noise_params = config.noise
        self.max_steps = config.rl.max_steps
        self.current_step = 0
        self.state = 0
        self.noise_history: deque = deque(maxlen=10)
        self.curriculum_episode = 0
    
    def set_curriculum_episode(self, episode: int):
        """Update curriculum learning progress."""
        self.curriculum_episode = episode
    
    def _get_phone_temp(self) -> float:
        """Simulate phone temperature with Gaussian distribution."""
        return max(25, np.random.normal(loc=35, scale=5))
    
    def _get_rf_level(self) -> float:
        """Simulate RF interference level with Beta distribution."""
        return np.random.beta(a=2, b=5)
    
    def _get_emi_level(self) -> float:
        """Simulate EMI level with Beta distribution."""
        return np.random.beta(a=3, b=4)
    
    def _compute_total_noise(self) -> float:
        """Compute total noise probability with correlated sources."""
        # Curriculum learning: scale noise based on training progress
        curriculum_factor = min(1.0, self.curriculum_episode / 100)
        
        temp = self._get_phone_temp()
        temp_error = self.noise_params.phone_temp_error_per_degree * \
                    max(0, temp - self.noise_params.phone_temp_base)
        
        rf_level = self._get_rf_level()
        emi_level = self._get_emi_level()
        
        rf_error = self.noise_params.phone_rf_error_rate * rf_level * curriculum_factor
        emi_error = self.noise_params.phone_emi_error_rate * emi_level * curriculum_factor
        
        thermal_error = (1 - np.exp(-self.noise_params.thermal_gate_time / 
                                   self.noise_params.thermal_t1)) + \
                       (1 - np.exp(-self.noise_params.thermal_gate_time / 
                                   self.noise_params.thermal_t2)) / 2
        
        base_error = self.noise_params.depolarizing_error_rate * curriculum_factor
        cov_rf_emi = self.noise_params.rf_emi_correlation * np.sqrt(rf_error * emi_error)
        
        noise_vector = np.array([base_error, temp_error, rf_error, emi_error, thermal_error])
        cov_matrix = np.diag(np.ones(len(noise_vector))) * 0.01
        cov_matrix[2, 3] = cov_matrix[3, 2] = cov_rf_emi
        
        correlated_noise = np.sqrt(np.sum(
            np.random.multivariate_normal(noise_vector, cov_matrix) ** 2
        ))
        total_noise = min(0.5, correlated_noise)
        
        self.noise_history.append(total_noise)
        return total_noise
    
    def reset(self) -> str:
        """Reset environment to a random error state."""
        self.current_step = 0
        self.state = random.randint(0, 31)  # 5 qubits = 2^5 states
        return self.get_syndrome()
    
    def get_syndrome(self, batch: Optional[List[int]] = None) -> Union[str, List[str]]:
        """Simulate syndrome measurement, supporting batch processing."""
        if batch is not None:
            return [self._get_single_syndrome(state) for state in batch]
        return self._get_single_syndrome(self.state)
    
    def _get_single_syndrome(self, state: int) -> str:
        """Helper for single syndrome generation with bitwise operations."""
        try:
            noise_prob = self._compute_total_noise()
            syndrome_int = self.quantum_code.get_syndrome(state)
            
            # Inject noise into syndrome using bitwise operations
            syndrome_bits = [int(b) for b in syndrome_int]
            for i in range(len(syndrome_bits)):
                if random.random() < noise_prob:
                    syndrome_bits[i] = 1 - syndrome_bits[i]  # Flip bit
            
            return ''.join(str(b) for b in syndrome_bits)
        except Exception as e:
            logger.error(f"Syndrome generation failed: {e}")
            return '0000'
    
    def step(self, action: int) -> Tuple[str, float, bool]:
        """Apply action and return next syndrome, reward, and done flag."""
        if not 0 <= action < self.config.rl.action_dim:
            raise ValueError(f"Invalid action: {action}. Must be in [0, {self.config.rl.action_dim-1}]")
        
        prev_errors = bin(self.state).count('1')
        
        if action < 5:  # Bit flip action
            self.state ^= (1 << action)  # Bitwise XOR for efficient flip
        
        new_syndrome = self.get_syndrome()
        errors = bin(self.state).count('1')
        
        # Reward function
        if new_syndrome == '0000':
            reward = 1.0
        elif prev_errors > 0:
            reward = 0.5 * (prev_errors - errors) / prev_errors
        else:
            reward = -0.1
        
        self.current_step += 1
        done = new_syndrome == '0000' or self.current_step >= self.max_steps
        
        return new_syndrome, reward, done

class MiniQECAgent:
    """MiniQEC agent with ParallelMiniDQN, TinyLSTM, and QAT support."""
    
    def __init__(self, config: MiniQECConfig = DEFAULT_CONFIG):
        self.config = config
        self.device = torch.device('cpu')
        
        # Initialize models with QAT support
        use_qat = config.hardware.quantization_aware_training
        
        self.dqn = ParallelMiniDQN(
            config.rl.state_dim, 
            config.rl.action_dim,
            config.rl.num_parallel_dqns,
            use_qat=use_qat
        ).to(self.device)
        
        self.target_dqn = ParallelMiniDQN(
            config.rl.state_dim, 
            config.rl.action_dim,
            config.rl.num_parallel_dqns,
            use_qat=use_qat
        ).to(self.device)
        
        self.target_dqn.load_state_dict(self.dqn.state_dict())
        
        # Adjust LSTM size based on resources
        hidden_dim = 4 if self._is_low_resource_device() else 8
        self.lstm = TinyLSTM(
            config.rl.state_dim, 
            hidden_dim,
            config.rl.bidirectional_lstm,
            use_qat=use_qat
        ).to(self.device)
        
        # Prepare for quantization if QAT is enabled
        if use_qat:
            self.dqn.fuse_model()
            self.target_dqn.fuse_model()
            self.dqn = quantization.prepare_qat(self.dqn)
            self.target_dqn = quantization.prepare_qat(self.target_dqn)
            self.lstm = quantization.prepare_qat(self.lstm)
        
        self.optimizer = torch.optim.Adam(self.dqn.parameters(), lr=config.rl.lr_dqn)
        self.lstm_optimizer = torch.optim.Adam(self.lstm.parameters(), lr=config.rl.lr_lstm)
        
        self.scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(
            self.optimizer, mode='max', factor=0.5, patience=20
        )
        self.lstm_scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(
            self.lstm_optimizer, mode='min', factor=0.5, patience=20
        )
        
        self.memory = PriorityReplayBuffer(config.rl.memory_capacity)
        self.batch_size = config.rl.batch_size
        self.gamma = config.rl.gamma
        self.epsilon = config.rl.epsilon_start
        self.epsilon_min = config.rl.epsilon_min
        self.epsilon_decay = config.rl.epsilon_decay
        self.per_alpha = config.rl.per_alpha
        self.per_beta = config.rl.per_beta
        self.per_beta_increment = config.rl.per_beta_increment
        self.alpha_lstm = config.rl.alpha_lstm_base
        self.sequence_length = config.rl.sequence_length_base
        
        self.syndrome_history: deque = deque(maxlen=self.sequence_length)
        self.step_counter = 0
        self.huber_loss = nn.HuberLoss()
        self.lstm_confidence = 0.5
        
        # Hardware accelerators
        self.fpga_enabled = config.hardware.fpga_enabled
        self.mock_fpga = MockFPGA(config.hardware) if self.fpga_enabled else None
        self.cpu_accelerator = CPUAccelerator()
        
        self.consensus = InternodesConsensus(config.rl.consensus_threshold)
        self.episode_rewards: List[float] = []
        self.model_hash: Optional[str] = None
    
    def _is_low_resource_device(self) -> bool:
        """Check if device has limited resources."""
        try:
            mem = psutil.virtual_memory()
            return mem.available < 2 * 1024 * 1024 * 1024  # 2GB
        except:
            return False
    
    def _state_to_tensor(self, state: str) -> torch.Tensor:
        """Convert syndrome string to tensor using bitwise operations."""
        return torch.tensor([float(s) for s in state], dtype=torch.float32).to(self.device)
    
    def _compute_td_error(self, state: str, action: int, reward: float, 
                         next_state: str, done: bool) -> float:
        """Compute TD error for prioritized experience replay."""
        state_tensor = self._state_to_tensor(state)
        next_state_tensor = self._state_to_tensor(next_state)
        
        q_value = self.dqn(state_tensor)[action]
        next_action = self.dqn(next_state_tensor).argmax().item()
        next_q_value = self.target_dqn(next_state_tensor)[next_action].item()
        
        target = reward + self.gamma * next_q_value * (1 - done)
        return abs(q_value.item() - target)
    
    async def choose_action_async(self, state: str, use_lstm: bool = True, 
                                  episode: int = 0, total_episodes: int = 500) -> int:
        """Select action asynchronously with FPGA support."""
        # Update LSTM blending weight
        self.alpha_lstm = min(0.5, self.config.rl.alpha_lstm_base + 
                             0.3 * (episode / total_episodes))
        
        if random.random() < self.epsilon:
            return random.randint(0, self.config.rl.action_dim - 1)
        
        state_tensor = self._state_to_tensor(state)
        
        if self.fpga_enabled and self.mock_fpga:
            return await self._fpga_inference_async(state_tensor)
        
        with torch.no_grad():
            if use_lstm and self.syndrome_history and \
               random.random() < min(0.9, self.lstm_confidence):
                lstm_input = torch.tensor(
                    [[float(s) for s in h] for h in self.syndrome_history],
                    dtype=torch.float32
                ).unsqueeze(0).to(self.device)
                
                pred_syndrome = self.lstm(lstm_input)
                current_q = self.dqn(state_tensor)
                pred_q_values = self.dqn(pred_syndrome)
                q_values = (1 - self.alpha_lstm) * current_q + self.alpha_lstm * pred_q_values
                return torch.argmax(q_values).item()
            else:
                return self.consensus.aggregate(
                    [sub_dqn(state_tensor) for sub_dqn in self.dqn.sub_dqns],
                    state_tensor, 
                    self.dqn.sub_dqns
                )
    
    def choose_action(self, state: str, use_lstm: bool = True, 
                     episode: int = 0, total_episodes: int = 500) -> int:
        """Synchronous wrapper for choose_action_async."""
        return asyncio.run(
            self.choose_action_async(state, use_lstm, episode, total_episodes)
        )
    
    async def _fpga_inference_async(self, state: torch.Tensor) -> int:
        """Perform async inference using mock FPGA."""
        if self.mock_fpga:
            q_values = await self.mock_fpga.infer(state, self.dqn)
            return torch.argmax(q_values).item()
        return self._cpu_inference(state)
    
    def _cpu_inference(self, state: torch.Tensor) -> int:
        """Perform CPU inference."""
        q_values = asyncio.run(self.cpu_accelerator.infer(state, self.dqn))
        return torch.argmax(q_values).item()
    
    def train_dqn(self, state: str, action: int, reward: float, 
                 next_state: str, done: bool) -> None:
        """Train ParallelMiniDQN with Double DQN and PER."""
        priority = self._compute_td_error(state, action, reward, next_state, done) + 1e-5
        self.memory.append(priority, (state, action, reward, next_state, done))
        
        if len(self.memory) < max(self.batch_size, self.config.rl.warmup_steps):
            return
        
        batch, weights, indices = self.memory.sample(
            self.batch_size, self.per_alpha, self.per_beta
        )
        
        states = torch.tensor(
            [[float(s) for s in t[0]] for t in batch], 
            dtype=torch.float32
        ).to(self.device)
        
        actions = torch.tensor([t[1] for t in batch], dtype=torch.long).to(self.device)
        rewards = torch.tensor([t[2] for t in batch], dtype=torch.float32).to(self.device)
        next_states = torch.tensor(
            [[float(s) for s in t[3]] for t in batch], 
            dtype=torch.float32
        ).to(self.device)
        
        dones = torch.tensor([t[4] for t in batch], dtype=torch.float32).to(self.device)
        
        q_values = self.dqn(states).gather(1, actions.unsqueeze(1)).squeeze(1)
        
        with torch.no_grad():
            next_actions = self.dqn(next_states).argmax(1)
            next_q_values = self.target_dqn(next_states).gather(
                1, next_actions.unsqueeze(1)
            ).squeeze(1)
            targets = rewards + self.gamma * next_q_values * (1 - dones)
        
        loss = self.huber_loss(q_values, targets) * \
               torch.tensor(weights, dtype=torch.float32).to(self.device).mean()
        
        self.optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(self.dqn.parameters(), max_norm=1.0)
        self.optimizer.step()
        
        td_errors = torch.abs(q_values - targets).detach().cpu().numpy()
        self.memory.update_priorities(indices, td_errors + 1e-5)
        
        self.per_beta = min(1.0, self.per_beta + self.per_beta_increment)
        self.step_counter += 1
        
        if self.step_counter % 50 == 0:
            self.target_dqn.load_state_dict(self.dqn.state_dict())
    
    def train_lstm(self, syndrome: str, next_syndrome: str) -> None:
        """Train LSTM to predict next syndrome."""
        self.syndrome_history.append(syndrome)
        
        if len(self.syndrome_history) < self.sequence_length:
            return
        
        lstm_input = torch.tensor(
            [[float(s) for s in h] for h in self.syndrome_history],
            dtype=torch.float32
        ).unsqueeze(0).to(self.device)
        
        target = torch.tensor(
            [float(s) for s in next_syndrome], 
            dtype=torch.float32
        ).unsqueeze(0).to(self.device)
        
        pred = torch.clamp(self.lstm(lstm_input), 1e-7, 1 - 1e-7)
        loss = nn.BCELoss()(pred, target)
        
        self.lstm_optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(self.lstm.parameters(), max_norm=1.0)
        self.lstm_optimizer.step()
        
        log_likelihood = -nn.BCELoss(reduction='none')(pred, target).mean().item()
        self.lstm_confidence = 1 / (1 + np.exp(-log_likelihood))
    
    def finalize_quantization(self):
        """Convert QAT models to quantized inference models."""
        if self.config.hardware.quantization_aware_training:
            self.dqn = quantization.convert(self.dqn)
            self.target_dqn = quantization.convert(self.target_dqn)
            self.lstm = quantization.convert(self.lstm)
            logger.info("Models converted to quantized inference mode")

class SecureModelStorage:
    """Secure model storage with encryption and integrity verification."""
    
    def __init__(self, key: Optional[bytes] = None):
        self.key = key or Fernet.generate_key()
        self.cipher = Fernet(self.key)
        self.model_hashes: Dict[str, str] = {}
    
    def save_encrypted_model(self, model: nn.Module, filename: str) -> str:
        """Save model weights encrypted with integrity hash."""
        with tempfile.NamedTemporaryFile(delete=False) as temp_file:
            torch.save(model.state_dict(), temp_file.name)
            
            with open(temp_file.name, 'rb') as f:
                encrypted_data = self.cipher.encrypt(f.read())
            
            # Compute hash before encryption for integrity
            model_hash = compute_model_hash(temp_file.name)
            self.model_hashes[filename] = model_hash
            
            with open(filename, 'wb') as f:
                # Store hash with encrypted data
                f.write(model_hash.encode() + b'\n')
                f.write(encrypted_data)
            
            os.unlink(temp_file.name)
        
        logger.info(f"Saved encrypted model to {filename} with hash {model_hash[:16]}...")
        return model_hash
    
    def load_encrypted_model(self, model: nn.Module, filename: str, 
                            expected_hash: Optional[str] = None) -> bool:
        """Load encrypted model with integrity verification."""
        with open(filename, 'rb') as f:
            stored_hash = f.readline().decode().strip()
            encrypted_data = f.read()
        
        # Verify integrity
        if expected_hash and stored_hash != expected_hash:
            logger.error(f"Hash mismatch. Expected: {expected_hash}, Got: {stored_hash}")
            return False
        
        with tempfile.NamedTemporaryFile(delete=False) as temp_file:
            temp_file.write(self.cipher.decrypt(encrypted_data))
            temp_file.flush()
            
            # Verify decrypted file hash
            decrypted_hash = compute_model_hash(temp_file.name)
            if decrypted_hash != stored_hash:
                logger.error("Decrypted model hash mismatch")
                os.unlink(temp_file.name)
                return False
            
            model.load_state_dict(torch.load(temp_file.name))
            os.unlink(temp_file.name)
        
        logger.info(f"Loaded encrypted model from {filename}")
        return True
    
    def get_key(self) -> bytes:
        """Get encryption key (store securely in production)."""
        return self.key

def monitor_system_resources(threshold_battery: float = 20.0, 
                            threshold_cpu: float = 80.0) -> bool:
    """Check battery and CPU usage to determine if training should pause."""
    try:
        battery = psutil.sensors_battery()
        cpu_percent = psutil.cpu_percent(interval=1)
        
        if battery and battery.percent < threshold_battery:
            logger.warning(f"Battery low: {battery.percent}%. Pausing training.")
            return True
        
        if cpu_percent > threshold_cpu:
            logger.warning(f"CPU usage high: {cpu_percent}%. Pausing training.")
            return True
        
        return False
    except Exception as e:
        logger.warning(f"Resource monitoring unavailable: {e}")
        return False

def train_miniqec(config: MiniQECConfig = DEFAULT_CONFIG, 
                 episodes: Optional[int] = None,
                 use_hardware: bool = False,
                 pretrained_path: Optional[str] = None) -> Optional[MiniQECAgent]:
    """Train MiniQEC agent with curriculum learning and early stopping."""
    
    if monitor_system_resources():
        logger.info("Insufficient resources. Pausing training.")
        return None
    
    episodes = episodes or config.rl.episodes
    env = QuantumPhoneEnv(config, use_hardware)
    agent = MiniQECAgent(config)
    
    # Secure storage
    storage = SecureModelStorage()
    encryption_key = storage.get_key()
    
    if pretrained_path and os.path.exists(pretrained_path):
        if storage.load_encrypted_model(agent.dqn, pretrained_path):
            agent.target_dqn.load_state_dict(agent.dqn.state_dict())
            logger.info(f"Loaded pretrained model from {pretrained_path}")
    
    rewards, losses, successes = [], [], []
    best_reward = -float('inf')
    patience_counter = 0
    
    # Metrics files
    metrics_file = 'training_metrics.csv'
    metrics_json = 'training_metrics.json'
    
    with open(metrics_file, 'w', newline='') as f:
        writer = csv.writer(f)
        writer.writerow(['Episode', 'Avg_Reward', 'Avg_Loss', 'Success_Rate'])
    
    with open(metrics_json, 'w') as f:
        json.dump([], f)
    
    for episode in range(episodes):
        if monitor_system_resources():
            logger.info("Resource constraints detected. Saving checkpoint.")
            storage.save_encrypted_model(agent.dqn, f'miniqec_checkpoint_{episode}.pth')
            return agent
        
        # Curriculum learning
        env.set_curriculum_episode(episode)
        noise_level = env._compute_total_noise()
        agent.sequence_length = min(
            config.rl.sequence_length_base + int(noise_level * 5), 
            10
        )
        agent.syndrome_history = deque(maxlen=agent.sequence_length)
        
        state = env.reset()
        total_reward = 0
        episode_losses = []
        done = False
        
        while not done:
            action = agent.choose_action(state, episode=episode, total_episodes=episodes)
            next_state, reward, done = env.step(action)
            
            agent.train_dqn(state, action, reward, next_state, done)
            agent.train_lstm(state, next_state)
            
            total_reward += reward
            episode_losses.append(
                agent._compute_td_error(state, action, reward, next_state, done)
            )
            state = next_state
        
        agent.episode_rewards.append(total_reward)
        rewards.append(total_reward)
        losses.append(np.mean(episode_losses) if episode_losses else 0)
        successes.append(1 if total_reward >= 0.5 else 0)
        
        agent.scheduler.step(total_reward)
        agent.lstm_scheduler.step(losses[-1])
        
        # Epsilon decay (per-episode)
        agent.epsilon = max(agent.epsilon_min, agent.epsilon * agent.epsilon_decay)
        
        # Logging every 50 episodes
        if episode % 50 == 0:
            avg_reward = np.mean(rewards[-50:]) if rewards else 0
            avg_loss = np.mean(losses[-50:]) if losses else 0
            success_rate = np.mean(successes[-50:]) * 100 if successes else 0
            
            logger.info(
                f"Episode {episode}, Avg Reward: {avg_reward:.2f}, "
                f"Avg Loss: {avg_loss:.4f}, Success Rate: {success_rate:.2f}%"
            )
            
            with open(metrics_file, 'a', newline='') as f:
                writer = csv.writer(f)
                writer.writerow([episode, avg_reward, avg_loss, success_rate])
            
            with open(metrics_json, 'r') as f:
                metrics_data = json.load(f)
            
            metrics_data.append({
                'episode': episode,
                'avg_reward': avg_reward,
                'avg_loss': avg_loss,
                'success_rate': success_rate
            })
            
            with open(metrics_json, 'w') as f:
                json.dump(metrics_data, f, indent=2)
            
            # Save model
            if avg_reward > best_reward:
                best_reward = avg_reward
                model_hash = storage.save_encrypted_model(
                    agent.dqn, 
                    'miniqec_dqn_best.pth'
                )
                agent.model_hash = model_hash
                logger.info(f"Saved best model with hash {model_hash[:16]}...")
                patience_counter = 0
            else:
                patience_counter += 1
            
            # Early stopping
            if patience_counter >= config.rl.early_stop_patience and \
               avg_reward >= config.rl.early_stop_min_reward:
                logger.info(f"Early stopping at episode {episode}")
                break
    
    # Finalize quantization for deployment
    agent.finalize_quantization()
    
    return agent

def test_miniqec(agent: MiniQECAgent, config: MiniQECConfig = DEFAULT_CONFIG,
                trials: Optional[int] = None, 
                use_hardware: bool = False,
                use_lstm: bool = True) -> Dict:
    """Test MiniQEC agent across dynamic noise levels."""
    
    trials = trials or config.rl.test_trials
    env = QuantumPhoneEnv(config, use_hardware)
    results = {}
    
    test_conditions = [
        {'name': 'low_noise', 'depolarizing': 0.01, 'temp': 25, 'rf': 0.1, 'emi': 0.1},
        {'name': 'med_noise', 'depolarizing': 0.05, 'temp': 35, 'rf': 0.5, 'emi': 0.5},
        {'name': 'high_noise', 'depolarizing': 0.1, 'temp': 45, 'rf': 1.0, 'emi': 1.0},
        {'name': 'dynamic_noise', 'depolarizing': None, 'temp': None, 'rf': None, 'emi': None},
        {'name': 'adversarial_noise', 'depolarizing': 0.5, 'temp': 50, 'rf': 1.0, 'emi': 1.0}
    ]
    
    for condition in test_conditions:
        successes = 0
        steps_list = []
        success_rates = []
        
        for _ in range(trials):
            if condition['depolarizing'] is not None:
                # Override noise params for this test
                env.noise_params = NoiseConfig(
                    depolarizing_error_rate=condition['depolarizing'],
                    phone_temp_base=condition['temp'],
                    phone_rf_error_rate=condition['rf'] * 0.02,
                    phone_emi_error_rate=condition['emi'] * 0.015
                )
            else:
                env.noise_params = NoiseConfig(
                    depolarizing_error_rate=np.random.uniform(0.01, 0.1),
                    phone_temp_base=np.random.uniform(25, 45)
                )
            
            state = env.reset()
            done = False
            steps = 0
            
            while not done:
                action = agent.choose_action(state, use_lstm=use_lstm)
                state, _, done = env.step(action)
                steps += 1
            
            if state == '0000':
                successes += 1
            
            steps_list.append(steps)
            success_rates.append(1 if state == '0000' else 0)
        
        success_rate = successes / trials * 100
        ler = 1 - success_rate / 100
        avg_steps = np.mean(steps_list) if steps_list else 0
        
        # Bootstrap confidence intervals
        bootstrap_samples = [
            np.mean(np.random.choice(success_rates, size=trials, replace=True)) 
            for _ in range(1000)
        ]
        ci_lower, ci_upper = np.percentile(bootstrap_samples, [2.5, 97.5]) * 100
        
        logger.info(
            f"{condition['name']}: Success Rate: {success_rate:.2f}% "
            f"[{ci_lower:.2f}-{ci_upper:.2f}], LER: {ler:.4f}, Avg Steps: {avg_steps:.2f}"
        )
        
        results[condition['name']] = {
            'success_rate': success_rate,
            'ler': ler,
            'avg_steps': avg_steps,
            'ci_lower': ci_lower,
            'ci_upper': ci_upper
        }
    
    return results

    assert q_values.shape[0] == config.rl.action_dim
    logger.info("✓ Agent model tests passed")

# ============================================================================

def export_to_onnx(agent: MiniQECAgent, output_path: str = "miniqec.onnx") -> bool:
    """Export trained model to ONNX for mobile deployment."""
    try:
        agent.dqn.eval()
        dummy_input = torch.randn(1, agent.config.rl.state_dim)
        
        torch.onnx.export(
            agent.dqn,
            dummy_input,
            output_path,
            input_names=['state'],
            output_names=['action'],
            dynamic_axes={
                'state': {0: 'batch_size'},
                'action': {0: 'batch_size'}
            },
            opset_version=11
        )
        
        logger.info(f"Model exported to ONNX: {output_path}")
        return True
    except Exception as e:
        logger.error(f"ONNX export failed: {e}")
        return False
